```python
# Saga orchestration for distributed transactions
class SagaOrchestrator:
    async def execute_saga(self, saga: SagaDefinition) -> SagaResult:
        executed_steps = []
        try:
            for step in saga.steps:
                result = await step.execute(context)
                executed_steps.append((step, result))
            return SagaResult(success=True)
        except Exception as e:
            await self.compensate_steps(executed_steps)
            return SagaResult(success=False, error=str(e))
```
